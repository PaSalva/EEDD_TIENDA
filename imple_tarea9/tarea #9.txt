
METODO METER AL CARRITO

public void meterAlCarrito(Articulo a) {
		this.carrito[iContadorCarrito] = a;
		this.iContadorCarrito++;
	}


METODO VENDER CARRITO

public String venderCarrito(Cliente c,Terminal t) {
		String sResultado = "";

		for (int i = 0; i <iContadorCarrito; i++) {
			t.getArtController().vender(carrito[i], c,t);
			sResultado += "Se ha vendido el producto " + carrito[i].getsNombre() + " \n";
		}

		return sResultado;
	}
